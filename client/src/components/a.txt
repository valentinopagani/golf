													const birdiePercentage = totalGolpes > 0 ? ((birdieCount / totalGolpes) * 100).toFixed(2) : '...';
													const bogeyPercentage = totalGolpes > 0 ? ((bogeyCount / totalGolpes) * 100).toFixed(2) : '...';

										onChange={(e) =>

												[hoyoId]: { ...prev[hoyoId], distancia: Number(e.target.value) }
